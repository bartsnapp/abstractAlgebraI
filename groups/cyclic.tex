\begin{definition}
  A group $G$ is \dfn{cyclic} if it is generated by one element,
  meaning
  \[
  \< a\> := \{a^n:n\in \Z\} = G.
  \]
\end{definition}


\begin{example}
  The group $(\Z,+)$ is a cyclic group.
\end{example}
