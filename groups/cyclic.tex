\begin{definition}
  A group $G$ is \dfn{cyclic} if it is generated by one element,
  meaning
  \[
  \< a\> := \{a^n:n\in \Z\} = G.
  \]
\end{definition}


\begin{example}
  The group $(\Z,+)$ is a cyclic group.
\end{example}


\begin{example}
  The group $(C,\cdot)$ where $C=\{1,i,-1,-i\}$ is a cyclic group.
\end{example}


\begin{exercise}
  Prove that every cyclic group is Abelian.
\end{exercise}


\begin{example}[Modular arithmetic]
  Let $\Z_{12}$ be arithmetic on a clock:
  \[
  \begin{tikzpicture} 
    \pic {tikznclock={12}};
  \end{tikzpicture}
  \]
  On this clock, if it is $3$ o'clock, and $13$ hours pass, is is now
  $4$ o'clock. In this case, we write
  \[
  3 + 13 \equiv 4 \pmod{12}.
  \]
  The group $(\Z_{12},+)$ is cyclic with generator $1$.
\end{example}
