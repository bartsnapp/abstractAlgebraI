\begin{definition}
  A subgroup $N\subgp G$ is a \dfn{normal} subgroup if
  \[
  aN = Na,
  \]
  meaning every left coset is equal to the right coset involving the
  same group element. In this case we write $N\normal G$. \index{normal@$\normal$}
\end{definition}

\begin{lemma}[Normal subgroup criterion]
  A subgroup $N\subgp G$ is normal if and only if for all $n\in N$ and
  all $g\in G$,
  \[
  gng^{-1}\in N.
  \]
  \begin{proof}
    $(\Rightarrow)$

    $(\Leftarrow)$
  \end{proof}
\end{lemma}
