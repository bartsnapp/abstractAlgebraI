\begin{abstract}
  We state the classification theorem for finite Abelian groups.
\end{abstract}
\maketitle


The next theorem is sometimes called the \textit{fundamental theorem
  of finite Abelian groups}:

\begin{theorem}[Finite Abelian groups]\index{fundamental theorem of finite Abelian groups}
  Every finite Abelian group is isomorphic to a direct product of
  cyclic groups.
  %% \begin{proof}
  %%   %%% https://math.stackexchange.com/questions/792528/simple-proof-of-the-structure-theorems-for-finite-abelian-groups
  %% \end{proof}
\end{theorem}



\begin{example}[Abelian groups of order $\boldsymbol{20}$]
  There are two isomorphism classes of Abelian groups of order $20$:
  \begin{itemize}
  \item $\Z_{20}\iso \Z_{5}\times \Z_4$
  \item $\Z_{10}\times\Z_2\iso \Z_5\times \Z_2\times \Z_2$.
  \end{itemize}
\end{example}
