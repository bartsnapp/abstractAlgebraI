\begin{abstract}
  We state the classification theorem for finite Abelian groups.
\end{abstract}
\maketitle

\begin{theorem}[Finite Abelian groups]
  Every Abelian group is isomorphic to a direct product of cyclic groups.
  %% \begin{proof}
  %%   %%% https://math.stackexchange.com/questions/792528/simple-proof-of-the-structure-theorems-for-finite-abelian-groups
  %% \end{proof}
\end{theorem}



\begin{example}[Abelian groups of order $\boldsymbol{20}$]
  There are two isomorphism classes of Abelian groups of order $10$:
  $\Z_{20}$ and $\Z_{10}\times\Z_2$.
\end{example}
